Cantidad de infectados y nuevas infecciones por da

df %>%
    group_by(FECHA_RESULTADO) %>%
    summarise(N = n()) %>%
    mutate(FECHA_RESULTADO = lubridate::dmy(FECHA_RESULTADO)) %>%
    arrange(FECHA_RESULTADO) %>%
    mutate(NSUM = cumsum(N)) -> df_infec

Funciones

Funcin para graficar un rango de los nuevos casos por da

plot_new_cases <- function(min_day, max_day) {
    df_infec %>%
        filter(FECHA_RESULTADO >= min_day & FECHA_RESULTADO <= max_day) %>%
        plot(N~FECHA_RESULTADO, data = .)
}

Funcin para graficar un rango de los casos totales

plot_sum_cases <- function(min_day, max_day) {
    df_infec %>%
        filter(FECHA_RESULTADO >= min_day & FECHA_RESULTADO <= max_day) %>%
        plot(NSUM~FECHA_RESULTADO, data = .)
}

Funcin para graficar un rango del cambio de movilizacin recreacional

plot_rmoves <- function(min_day, max_day) {
    df_mov %>%
        group_by(date) %>%
        summarise(mov = mean(retail_and_recreation_percent_change_from_baseline, na.rm = TRUE)) %>%
        filter(date >= min_day & date <= max_day) %>%
        plot()
}

DRY

cat_function <- function(f_name, min_day, max_day) {
    cat("```{r}\n")
    cat(f_name)
    cat("(\"")
    cat(min_day)
    cat("\", \"")
    cat(max_day)
    cat("\")\n```\n\n")
}

cat_title <- function(f_title, min_day, max_day) {
    cat("##",
        f_title,
        format(as.Date(min_day), "%m-%d"),
        "->",
        format(as.Date(max_day), "%m-%d"),
        "\n"
    )
}

repeat_plots <- function(min_day, max_day) {
    cat_title("Nuevos casos", min_day, max_day)
    cat_function("plot_new_cases", min_day, max_day)
    plot_new_cases(min_day, max_day)
    cat(".\n\n\n")

    cat_title("Casos totales", min_day, max_day)
    cat_function("plot_sum_cases", min_day, max_day)
    plot_sum_cases(min_day, max_day)
    cat(".\n\n\n")

    cat_title("Movilizacin recreacional", min_day, max_day)
    cat_function("plot_rmoves", min_day, max_day)
    plot_rmoves(min_day, max_day)
    cat(".\n\n\n")
}
